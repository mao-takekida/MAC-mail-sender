-- メールの件名と内容を設定set emailSubject to "件名" -- ここに件名set emailContent to "内容" -- ここに内容-- 受信者のメールアドレスを設定set toEmail to "hogehoge@example.com" -- Toが不要な場合は空文字列set ccEmail to "" -- CCが不要な場合は空文字列set bccEmail to "" -- BCCが不要な場合は空文字列-- 確認ダイアログを表示set dialogText to "以下の内容でメールを送信しますか？" & return & return & "件名: " & emailSubject & return & "内容: " & emailContentdisplay dialog dialogText buttons {"キャンセル", "送信"} default button "送信"-- ユーザーが「送信」をクリックした場合のみメールを送信if button returned of result is "送信" then	-- Outlookを使用してメールを作成・送信	tell application "Microsoft Outlook"		set newMessage to make new outgoing message with properties {subject:emailSubject, content:emailContent}				-- To の受信者を設定		if toEmail is not equal to "" then			make new recipient at end of to recipients of newMessage with properties {email address:{address:toEmail}}		end if				-- CC の受信者を設定		if ccEmail is not equal to "" then			make new recipient at end of cc recipients of newMessage with properties {email address:{address:ccEmail}}		end if				-- BCC の受信者を設定		if bccEmail is not equal to "" then			make new recipient at end of bcc recipients of newMessage with properties {email address:{address:bccEmail}}		end if				-- メールを送信		send newMessage	end tellelse	display dialog "メールの送信がキャンセルされました。" buttons {"OK"} default button "OK"end if
